```python
#### 백준 알고리즘 
# -별찍기2
#첫쨰 줄에는 별 1개 둘쨰 줄에는 별2개 n번쨰 줄에는 별n개를 찍는 문제
#입력:첫번째 줄에 (1<=n<=100이 주어진다)
# 출력:첫쨰 줄부터 n번째줄까지 차례대로 별을 출력한다


n=int(input())

for i in range(n+1):
    i+1
    print((i*"*").rjust(5))

# 오른쪽 정렬하는 함수: rjust(자릿수)을 이용한다
```
